// Inputs from vertex shader
in vec3 FragColor;
in vec2 FragTexCoords;

// Uniform Inputs
uniform sampler2D Texture0;
uniform sampler2D Texture1;

// get current time
uniform float CurrentTime;

// Output
out vec4 FinalColor;

void main()
{
	FinalColor = mix(texture(Texture0, FragTexCoords), texture(Texture1, FragTexCoords), 1.0f) /* vec4(FragColor, 1.0f)*/;
}
