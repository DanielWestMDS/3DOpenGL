#version 460 core

// vertex shader interpretations
layout (location = 0) in vec3 Position;
layout (location = 1) in vec2 TexCoords; 

// inputs
uniform mat4 QuadModelMat;
uniform mat4 ProjectionMat;
uniform mat4 ViewMat;

uniform mat4 MVPs[256];

// outputs to fragment shader
out vec2 FragTexCoords; // Only pass through texture coordinates

// shader functionality
void main()
{
    gl_Position = ProjectionMat * ViewMat * QuadModelMat * vec4(Position, 1.0f);
    FragTexCoords = TexCoords; // Pass through texture coordinates
}
